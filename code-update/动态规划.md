## 1007 Maximum Subsequence Sum (25分)
```c++
#include <iostream>
#include <algorithm>

using namespace std;

const int N = 100010;
int g[N];
int n;

int main(void) {
	cin >> n;
	for (int i = 1; i <= n; i++)
		cin >> g[i];

	int res = -1, cur = -1;
	int  left = 0, right = 0;
	for (int i = 1, start = -1; i <= n; i++) {
		if (cur < 0) cur = 0, start = i;
		cur += g[i];
		if (res < cur) {
			res = cur;
			left = g[start], right = g[i];
		}
	}
	if (res < 0) res = 0, left = g[1], right = g[n];

	cout << res << " " << left << " " << right << endl;
	return 0;
}
```

## 1045 Favorite Color Stripe (30分)
```c++
#include <iostream>
#include <algorithm>

using namespace std;

const int N = 210, M = N, L = 10010;
int n, m, l;
int p[M], q[L], f[M][L];

int main(void) {
	cin >> n >> m;
	for (int i = 1; i <= m; i++)
		cin >> p[i];
	cin >> l;
	for (int i = 1; i <= l; i++)
		cin >> q[i];
	
	for (int i = 1; i <= m; i++) {
		for (int j = 1; j <= l; j++) {
			f[i][j] = max(f[i][j - 1], f[i - 1][j]);
			if(p[i] == q[j]) f[i][j] = max(f[i][j], f[i][j - 1] + 1);
		}
	}
	cout << f[m][l] << endl;

	return 0;
}
```

## 1093 Count PAT's (25分)
```c++
#include <iostream>

using namespace std;

const int N = 100010, R = 1000000007;
int l[N], r[N];
char p[N];

int main(void) {
	string s;
	cin >> s;
	for (int i = 0; i < s.size() - 1; i++) {
		if (s[i] == 'P')
			l[i + 1] = l[i] + 1;
		else
			l[i + 1] = l[i];
	}
	for (int i = s.size() - 1; i > 0; i--) {
		if (s[i] == 'T')
			r[i - 1] = r[i] + 1;
		else
			r[i - 1] = r[i];
	}

	int res = 0;
	for (int i = 1; i < s.size() - 1; i++) {
		if (s[i] == 'A') {
			res += l[i] * r[i];
			res %= R;
		}
	}
	cout << res << endl;

	return 0;
}
```

## 1101 Quick Sort (25分)
```c++
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

const int N = 100010;
int g[N], l[N], r[N];
int n;

int main(void) {
	cin >> n;
	for (int i = 1; i <= n; i++)
		cin >> g[i];
	r[0] = 0, r[n + 1] = 0x3f3f3f3f;

	for (int i = 1; i <= n; i++)
		l[i] = max(l[i - 1], g[i]);
	for (int i = n; i >= 1; i--)
		r[i] = min(r[i + 1], g[i]);

	vector<int> res;
	
	for (int i = 1; i <= n; i++)
		if (g[i] >= l[i - 1] && g[i] <= r[i + 1])
			res.push_back(g[i]);

	sort(res.begin(), res.end());

	cout << res.size() << endl;
	if (res.empty())
		cout << endl;
	else {
		cout << res[0];
		for (int i = 1; i < res.size(); i++)
			cout << " " << res[i];
		cout << endl;
	}
	return 0;
}
```
