## 1013 Battle Over Cities (25分)
```c++
#include <iostream>

using namespace std;

const int N = 1010, M = N * (N - 1) / 2;
int p[N];
struct Edge {
	int a, b;
}edges[M];

int n, m, k;

int find(int x) {
	if (p[x] != x) p[x] = find(p[x]);
	return p[x];
}

int main(void) {
    scanf("%d%d%d", &n, &m, &k);
	for (int i = 0; i < m; i++) {
        scanf("%d%d", &edges[i].a, &edges[i].b);
	}
	while (k--) {
		int v;
		scanf("%d", &v);
		for (int i = 1; i <= n; i++)
			p[i] = i;
		int res = n - 1;
		for (int i = 0; i < m; i++) {
			int x = edges[i].a, y = edges[i].b;
			if (x == v || y == v) continue;
			if (find(x) != find(y)) {
				p[find(x)] = find(y);
				res--;
			}
		}
		printf("%d\n", res - 1);
	}
	return 0;
}
```

## 1114 Family Property (25分)
```c++
#include <iostream>
#include <algorithm>

using namespace std;

const int N = 10010;
int p[N], cnt[N];
bool st[N];
double estate[N], area[N];
int n, m;

struct Edge {
	int a, b;
}edges[N];

struct Family {
	int id, cnt;
	double avg_sets, avg_area;
	bool operator< (const Family& f) {
		if (avg_area != f.avg_area) return avg_area > f.avg_area;
		return id < f.id;
	}
}family[N];

int find(int x) {
	if (p[x] != x) p[x] = find(p[x]);
	return p[x];
}

int main(void) {
	for (int i = 0; i < N; i++) {
		p[i] = i;
		cnt[i] = 1;
	}
	int id, father, mother, k, child;
	double es, ar;
	cin >> n;
	for (int i = 0; i < n; i++) {
		cin >> id >> father >> mother >> k;
		st[id] = true;
		if(father != -1) edges[m++] = { id, father }, st[father] = true;
		if(mother != -1) edges[m++] = { id, mother }, st[mother] = true;
		for (int j = 0; j < k; j++) {
			cin >> child;
			edges[m++] = { id, child };
			st[child] = true;
		}
		cin >> es >> ar;
		estate[id] = es, area[id] = ar;
	}
	for (int i = 0; i < m; i++) {
		int pa = edges[i].a, qb = edges[i].b;
		int fp = find(pa), fq = find(qb);
		if (fp != fq) {
			if (fp > fq) swap(fp, fq);
			estate[fp] += estate[fq];
			area[fp] += area[fq];
			cnt[fp] += cnt[fq];
			p[fq] = fp;
		}
	}
	int count = 0;
	for (int i = 0; i < N; i++) {
		if (p[i] == i && st[i]) {
			double avg_sets = estate[i] / cnt[i];
			double avg_area = area[i] / cnt[i];
			family[count++] = { i, cnt[i], avg_sets, avg_area };
		}
	}
	sort(family, family + count);
	printf("%d\n", count);
	for (int i = 0; i < count; i++) {
		printf("%04d %d %.3lf %.3lf\n", family[i].id, family[i].cnt, family[i].avg_sets, family[i].avg_area);
	}
	return 0;
}
```

## 1118 Birds in Forest (25分)
```c++
#include <iostream>
#include <algorithm>

using namespace std;

const int N = 10010;
int p[N], birds[N];
int n, k, m, t, q;

int find(int x) {
	if (p[x] != x) p[x] = find(p[x]);
	return p[x];
}

int main(void) {
	for (int i = 0; i < N; i++)
		p[i] = i;
	cin >> n;
	int cnt = 0;
	for (int i = 0; i < n; i++) {
		int k;
		cin >> k;
		for (int j = 0; j < k; j++) {
			cin >> birds[j];
			m = max(m, birds[j]);
		}
		
		for (int j = 1; j < k; j++) {
			int a = birds[j-1], b = birds[j];
			int pa = find(a), pb = find(b);
			if (pa != pb) {
				p[pa] = pb;
				cnt++;
			}
		}
	}		
	cout << m - cnt << " " << m << endl;

	cin >> q;
	while (q--) {
		int a, b;
		cin >> a >> b;
		if (find(a) == find(b)) puts("Yes");
		else puts("No");
	}
	return 0;
}
```

## 1107 Social Clusters (30分)
```c++
#include <iostream>
#include <unordered_map>
#include <vector>
#include <algorithm>

using namespace std;

const int N = 1010;
int p[N], cnt[N];
unordered_map<int, vector<int>> hobby;
int n;

int find(int x) {
	if (p[x] != x) p[x] = find(p[x]);
	return p[x];
}

int main(void) {
	cin >> n;
	for (int i = 1; i <= n; i++) {
		p[i] = i;
		cnt[i] = 1;
	}
	for (int i = 1; i <= n; i++) {
		int k;
		scanf("%d:", &k);
		for (int j = 0; j < k; j++) {
			int v;
			cin >> v;
			hobby[v].push_back(i);
		}
	}
	for (auto& h : hobby) {
		auto& ph = h.second;
		if (ph.size() < 2) continue;
		for (int i = 1; i < ph.size(); i++) {
			int pa = find(ph[i - 1]), pb = find(ph[i]);
			if (pa != pb) {
				cnt[pa] += cnt[pb];
				p[pb] = pa;
			}
		}
	}
	vector<int> res;
	for (int i = 1; i <= n; i++)
		if (p[i] == i)
			res.push_back(cnt[i]);
	sort(res.rbegin(), res.rend());
	
	cout << res.size() << endl;
	cout << res[0];
	for (int i = 1; i < res.size(); i++)
		cout << " " << res[i];
	cout << endl;

	return 0;
}
```
