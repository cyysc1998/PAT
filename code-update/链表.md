```c++
#include <iostream>

	int x = h1, y = h2;
```

## 1074 Reversing Linked List (25分)
```c++
#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

const int N = 100010;

int h, e[N], ne[N];
int n, k;
vector<int> nodes;

int main(void) {
	scanf("%d%d%d", &h, &n, &k);
	for (int i = 0; i < n; i++) {
		int addr, v, next;
		scanf("%d%d%d", &addr, &v, &next);
		e[addr] = v, ne[addr] = next;
	}
    
	for (int i = h; i != -1; i = ne[i])
		nodes.push_back(i);
    
	for (int i = 0; i + k <= nodes.size(); i += k) {
		reverse(nodes.begin() + i, nodes.begin() + i + k);
	}

	for (int i = 0; i < nodes.size(); i++) {
		if (i != nodes.size() - 1)
			printf("%05d %d %05d\n", nodes[i], e[nodes[i]], nodes[i + 1]);
		else
			printf("%05d %d -1\n", nodes[i], e[nodes[i]]);
	}

	return 0;
}
```

## 1097 Deduplication on a Linked List (25分)
```c++
#include <iostream>
#include <vector>
#include <unordered_map>
#include <cmath>

using namespace std;

const int N = 100010;
int h, e[N], ne[N];
int n;

int main(void) {
	cin >> h >> n;
	for (int i = 0; i < n; i++) {
		int addr, v, next;
		cin >> addr >> v >> next;
		e[addr] = v, ne[addr] = next;
	}

	vector<int> p, q;
	unordered_map<int, bool> st;

	for (int i = h; i != -1; i = ne[i]) {
		if(st[abs(e[i])])
			q.push_back(i);
		else {
			p.push_back(i);
			st[abs(e[i])] = true;
		}		
	}

	for (int i = 0; i < p.size(); i++) {
		if (i != p.size() - 1)
			printf("%05d %d %05d\n", p[i], e[p[i]], p[i + 1]);
		else
			printf("%05d %d -1\n", p[i], e[p[i]]);
	}
	for (int i = 0; i < q.size(); i++) {
		if (i != q.size() - 1)
			printf("%05d %d %05d\n", q[i], e[q[i]], q[i + 1]);
		else
			printf("%05d %d -1\n", q[i], e[q[i]]);
	}
	
	return 0;
}
```

## 1133 Splitting A Linked List (25分)
```c++
#include <iostream>
#include <vector>

using namespace std;

const int N = 100010;
int h, e[N], ne[N];
int n, k;

int main(void) {
	scanf("%d%d%d", &h, &n, &k);
	for (int i = 0; i < n; i++) {
		int addr, v, next;
		scanf("%d%d%d", &addr, &v, &next);
		e[addr] = v, ne[addr] = next;
	}

	vector<int> q;

	for (int i = h; i != -1; i = ne[i])
		q.push_back(i);

	vector<int> p;

	for (int i = 0; i < q.size(); i++) {
		if (e[q[i]] < 0)
			p.push_back(q[i]);
	}

	for (int i = 0; i < q.size(); i++) {
		if (e[q[i]] >= 0)
			p.push_back(q[i]);
	}

	vector<int> t;

	for (int i = 0; i < p.size(); i++) {
		if (e[p[i]] <= k)
			t.push_back(p[i]);
	}

	for (int i = 0; i < p.size(); i++) {
		if (e[p[i]] > k)
			t.push_back(p[i]);
	}
	
	for (int i = 0; i < t.size(); i++) {
		if (i != t.size() - 1)
			printf("%05d %d %05d\n", t[i], e[t[i]], t[i + 1]);
		else
			printf("%05d %d -1\n", t[i], e[t[i]]);
	}

	return 0;
}
```
